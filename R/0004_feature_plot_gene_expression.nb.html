rownames(gene_counts) <- gene_counts$V1
gene_counts <- gene_counts[,-1]
gene_counts <- gene_counts %>% rownames_to_column("gene") %>% pivot_longer(
  cols = colnames(gene_counts),
  names_to = "sample",
  values_to = "counts"
)
